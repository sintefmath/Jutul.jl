================================================================================
EXAMPLE USAGE
================================================================================

using Jutul

# Create a simple 2D PEBI mesh with 4 points
points = [0.0 1.0 0.0 1.0; 
          0.0 0.0 1.0 1.0]

mesh = Jutul.VoronoiMeshes.PEBIMesh2D(points)

# Mesh properties
println(number_of_cells(mesh))  # Output: 4
println(length(mesh.node_points))  # Output: 9

# Access mesh structure
for cell_idx in 1:number_of_cells(mesh)
    faces = mesh.faces.cells_to_faces[cell_idx]
    println("Cell $cell_idx: faces $faces")
end
